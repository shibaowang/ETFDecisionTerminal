find_package(Qt6 COMPONENTS Core REQUIRED)

add_library(ShellAccountingTradeDraftConfirmationImplementationLib STATIC
    ShellAccountingTradeDraftConfirmationImplementation.cpp
)
target_include_directories(ShellAccountingTradeDraftConfirmationImplementationLib
    PUBLIC
        ${CMAKE_CURRENT_SOURCE_DIR}
        ${PROJECT_SOURCE_DIR}/tests/ShellAccountingTradeDraftImplementation
)
target_link_libraries(ShellAccountingTradeDraftConfirmationImplementationLib
    PUBLIC
        ShellAccountingTradeDraftImplementationLib
        ETFDecisionTerminal::DataServiceApi
        Qt6::Core
)
target_compile_features(ShellAccountingTradeDraftConfirmationImplementationLib PUBLIC cxx_std_17)

function(add_shell_accounting_tradedraft_confirmation_test target_name source_name ctest_name)
    add_executable(${target_name} ${source_name})
    target_link_libraries(${target_name}
        PRIVATE
            ShellAccountingTradeDraftConfirmationImplementationLib
    )
    target_compile_features(${target_name} PRIVATE cxx_std_17)
    add_test(
        NAME ${ctest_name}
        COMMAND
            ${target_name}
            --source-root ${PROJECT_SOURCE_DIR}
            --migration ${PROJECT_SOURCE_DIR}/migrations/001_initial_schema.sql
    )
    set_tests_properties(${ctest_name} PROPERTIES
        ENVIRONMENT_MODIFICATION
            "PATH=path_list_prepend:$<TARGET_FILE_DIR:Qt6::Core>"
    )
endfunction()
