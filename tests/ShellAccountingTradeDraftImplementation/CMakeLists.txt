add_library(ShellAccountingTradeDraftImplementationLib STATIC
    ShellAccountingTradeDraftImplementation.cpp
)
target_include_directories(ShellAccountingTradeDraftImplementationLib
    PUBLIC
        ${CMAKE_CURRENT_SOURCE_DIR}
        ${PROJECT_SOURCE_DIR}/tests/ShellAccountingAuditWriteImplementation
)
target_link_libraries(ShellAccountingTradeDraftImplementationLib
    PUBLIC
        ShellAccountingAuditWriteImplementationLib
        ETFDecisionTerminal::DataServiceApi
        Qt6::Core
)
target_compile_features(ShellAccountingTradeDraftImplementationLib PUBLIC cxx_std_17)

function(add_shell_accounting_tradedraft_test target_name source_name ctest_name)
    add_executable(${target_name} ${source_name})
    target_link_libraries(${target_name}
        PRIVATE
            ShellAccountingTradeDraftImplementationLib
    )
    target_compile_features(${target_name} PRIVATE cxx_std_17)
    add_test(
        NAME ${ctest_name}
        COMMAND
            ${target_name}
            --source-root ${PROJECT_SOURCE_DIR}
            --migration ${PROJECT_SOURCE_DIR}/migrations/001_initial_schema.sql
    )
    set_tests_properties(${ctest_name} PROPERTIES
        ENVIRONMENT_MODIFICATION
            "PATH=path_list_prepend:$<TARGET_FILE_DIR:Qt6::Core>"
    )
endfunction()
